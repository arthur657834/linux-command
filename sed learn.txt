a在当前行后面加入一行文本。 
i在当前行上面插入文本
#匹配行前加
sed -i '/allow 361way.com/iallow www.361way.com' the.conf.file
#匹配行前后
sed -i '/allow 361way.com/aallow www.361way.com' the.conf.file

插入文件第二行

sed '1d' ab
#删除第一行
sed '2,$d' ab
#删除第二行到最后一行

sed -n '1p' ab
#显示第一行 

sed -n '/ruby/p' ab
#查询包括关键字ruby所在所有行

sed '1,3a drink tea' ab
#第一行到第三行后增加字符串"drink tea"

sed '1a drink tea\nor coffee' ab
#第一行后增加多行，使用换行符\n

sed '1,2c Hi' ab
#第一行到第二行代替为Hi

sed -i 's/ruby/bird/g'
#替换ruby为bird

sed -i '$a bye' ab
#在文件ab中最后一行直接输入"bye"

sed -i '/匹配字符串/d'  filename
#删除匹配行

sed -i 's/替换源字符串/替换目标字符串/g' filename
#替换匹配行中的某个字符串

sed -e '/test/h' -e '/check/x' example
#包含test与check的行互换

sed '1,10y/abcde/ABCDE/' example 
#把1--10行内所有abcde转变为大写
